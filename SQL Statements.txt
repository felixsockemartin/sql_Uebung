DROP TABLE IF EXISTS Ort, Mannschaft, M_P, Person, Rolle, MP_R, Ergebnis, Spiel, M_S;
CREATE TABLE Ort (O_ID INT PRIMARY KEY, Name TEXT NOT NULL);
CREATE TABLE Mannschaft (M_ID INT PRIMARY KEY, Name TEXT NOT NULL, Ort INT NOT NULL, Punkte INT);
CREATE TABLE M_P (M_ID INT NOT NULL, P_ID INT NOT NULL);
CREATE TABLE Person (P_ID INT PRIMARY KEY, V_Name TEXT NOT NULL, N_Name TEXT NOT NULL, Geb_Tag CHAR(50), Geschlecht TEXT);
CREATE TABLE MP_R (P_ID INT NOT NULL, R_ID INT NOT NULL);
CREATE TABLE Rolle (R_ID INT PRIMARY KEY, Name TEXT NOT NULL);
CREATE TABLE M_S (M_ID INT NOT NULL, S_ID INT NOT NULL);
CREATE TABLE Spiel (S_ID INT PRIMARY KEY, Gastgeber INT NOT NULL, Datum CHAR(50) NOT NULL);
CREATE TABLE Ergebnis (E_ID INT PRIMARY KEY, S_ID INT NOT NULL, P_ID INT NOT NULL, Min INT NOT NULL, M_ID INT NOT NULL);
